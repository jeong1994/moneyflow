CREATE TABLE USERS (
  USER_NUMBER NUMBER,		-- PK
  EMAIL VARCHAR2(100) NOT NULL UNIQUE,
  PASSWORD VARCHAR2(100) NOT NULL,
  PHONE VARCHAR2(20),
  NICKNAME VARCHAR2(50) NOT NULL UNIQUE
);

CREATE TABLE CATEGORIES (
  CATEGORIE_NUMBER,
  USER_NUMBER NUMBER,         -- FK to USERS
  CATEGORY_NAME VARCHAR2(50),
  CATEGORY_SCOPE VARCHAR2(10) CHECK (CATEGORY_SCOPE IN ('기본', '커스텀')),
  CATEGORY_TYPE VARCHAR2(10) CHECK (CATEGORY_TYPE IN ('수입', '지출')),
);

CREATE TABLE TRANSACTIONS (
  TRANSACTION_ID NUMBER, 	-- PK
  USER_ID NUMBER,		-- FK to USERS
  CATEGORY_NUMBER NUMBER,	-- FK to CATEGORIES
  TRANSACTION_DATE DATE NOT NULL,
  AMOUNT NUMBER(10) NOT NULL,
  MEMO VARCHAR2(100)
);
